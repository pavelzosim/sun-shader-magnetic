precision highp float; uniform float time; uniform float umbraThresh; uniform float penumbraThresh; uniform float fibrilIntensity; varying vec2 vUv; float hash(vec2 p) { return fract(sin(dot(p, vec2(12.9898, 78.233))) * 43758.5453); } void main() { vec2 uv = (vUv - 0.5) * 2.0; float r = length(uv); // 1. Bz Field - Магнитная интенсивность float Bz = smoothstep(1.0, 0.0, r); // 2. Flow & Angle - вычисление направления волокон через atan2 // В оригинале используется запеченный FlowVec, здесь мы имитируем его радиально float baseAngle = atan(uv.y, uv.x); // VortexStrength - вихревое закручивание, нарастающее к центру float angle = baseAngle + (2.5 * fibrilIntensity) * (Bz * Bz); // 3. Multi-layer Fibril Compositing (Z-buffer logic) float bestStrandHeight = 0.0; float bestFibers = 0.0; float bestShading = 0.0; float FibrilFreq = 60.0; float safeTime = mod(time, 3600.0); // Оптимизированный цикл из 3 итераций как в оригинале for (int i = 0; i < 3; i++) { float lf = float(i); // Сдвиг секторов для переплетения нитей[cite: 1] float lSectorF = (angle + lf * 0.31) * (FibrilFreq / 6.2831); float lAngleID = floor(lSectorF); // Индивидуальные параметры нити через хэш[cite: 1] float lr1 = hash(vec2(lAngleID, lf * 17.13)); float lr2 = hash(vec2(lAngleID, lf * 43.71)); // Фазовый дрифт (анимация течения плазмы)[cite: 1] float lPhase = lr1 * 6.2831 + safeTime * ((lr2 * 2.0 - 1.0) * 0.5); // Цилиндрический профиль (имитация трубки)[cite: 1] float lRaw = cos(angle * FibrilFreq + lPhase); float lCylinder = sqrt(max(0.0, 1.0 - lRaw * lRaw)); // Вычисление Z-уровня для окклюзии[cite: 1] float lBaseZ = 1.0 - lf * 0.28; float lStrandHeight = lCylinder * lBaseZ; // Z-buffer select: побеждает тот, кто выше[cite: 1] if (lStrandHeight > bestStrandHeight) { bestStrandHeight = lStrandHeight; bestFibers = pow(lCylinder, 3.0) * (0.55 + lr1 * 0.9); // FibrilSharpness[cite: 1] bestShading = mix(0.45, 1.0, pow(lCylinder, 0.5)); // Self-shadowing[cite: 1] } } // 4. Маски зон (Penumbra & Umbra)[cite: 1] float penW = smoothstep(penumbraThresh * 0.85, penumbraThresh + 0.22, Bz); float umbra = smoothstep(umbraThresh, 1.0, Bz); float penumbraZone = clamp(penW * (1.0 - umbra), 0.0, 1.0); // 5. Финальная сборка цвета[cite: 1] vec3 baseColor = vec3(1.0, 0.55, 0.1); // Фотосфера vec3 penColor = vec3(0.4, 0.18, 0.05); // База пенумбры vec3 umbColor = vec3(0.01, 0.005, 0.002); // Умбра // Эффект нитей (fiberEffect)[cite: 1] float fiberEffect = bestFibers * (1.0 - 0.6) * (1.0 - penW) * (1.0 - umbra); // Смешивание по логике "darkeningWithFibers"[cite: 1] vec3 finalColor = mix(baseColor, penColor, penW); // Наложение теней между нитями[cite: 1] finalColor = mix(finalColor, finalColor * bestShading, penumbraZone * 0.8); finalColor += fiberEffect * 0.3; // Добавочная яркость волокон[cite: 1] // Умбра перекрывает всё сверху[cite: 1] finalColor = mix(finalColor, umbColor, umbra); gl_FragColor = vec4(finalColor, 1.0); }
